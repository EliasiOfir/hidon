# גניבת הברכות
## פרק: כ"ז

## טקסט מקורי

(א) וַיְהִי כִּי זָקֵן יִצְחָק וַתִּכְהֶיןָ עֵינָיו מֵרְאֹת וַיִּקְרָא אֶת עֵשָׂו בְּנוֹ הַגָּדֹל וַיֹּאמֶר אֵלָיו בְּנִי וַיֹּאמֶר אֵלָיו הִנֵּנִי:
(ב) וַיֹּאמֶר הִנֵּה נָא זָקַנְתִּי לֹא יָדַעְתִּי יוֹם מוֹתִי:
(ג) וְעַתָּה שָׂא נָא כֵלֶיךָ תֶּלְיְךָ וְקַשְׁתֶּךָ וְצֵא הַשָּׂדֶה וְצוּדָה לִּי צֵידָה:
(ד) וַעֲשֵׂה לִי מַטְעַמִּים כַּאֲשֶׁר אָהַבְתִּי וְהָבִיאָה לִּי וְאֹכֵלָה בַּעֲבוּר תְּבָרֶכְךָ נַפְשִׁי בְּטֶרֶם אָמוּת:
(ה) וְרִבְקָה שֹׁמַעַת בְּדַבֵּר יִצְחָק אֶל עֵשָׂו בְּנוֹ וַיֵּלֶךְ עֵשָׂו הַשָּׂדֶה לָצוּד צַיִד לְהָבִיא:
(ו) וְרִבְקָה אָמְרָה אֶל יַעֲקֹב בְּנָהּ לֵאמֹר הִנֵּה שָׁמַעְתִּי אֶת אָבִיךָ מְדַבֵּר אֶל עֵשָׂו אָחִיךָ לֵאמֹר:
(ז) הָבִיאָה לִּי צַיִד וַעֲשֵׂה לִי מַטְעַמִּים וְאֹכֵלָה וַאֲבָרֶכְכָה לִפְנֵי יְהוָה לִפְנֵי מוֹתִי:
(ח) וְעַתָּה בְנִי שְׁמַע בְּקֹלִי לַאֲשֶׁר אֲנִי מְצַוָּה אֹתָךְ:
(ט) לֶךְ נָא אֶל הַצֹּאן וְקַח לִי מִשָּׁם שְׁנֵי גְּדָיֵי עִזִּים טֹבִים וְאֶעֱשֶׂה אֹתָם מַטְעַמִּים לְאָבִיךָ כַּאֲשֶׁר אָהֵב:
(י) וְהֵבֵאתָ לְאָבִיךָ וְאָכָל בַּעֲבֻר אֲשֶׁר יְבָרֶכְךָ לִפְנֵי מוֹתוֹ:
(יא) וַיֹּאמֶר יַעֲקֹב אֶל רִבְקָה אִמּוֹ הֵן עֵשָׂו אָחִי אִישׁ שָׂעִר וְאָנֹכִי אִישׁ חָלָק:
(יב) אוּלַי יְמֻשֵּׁנִי אָבִי וְהָיִיתִי בְעֵינָיו כִּמְתַעְתֵּעַ וְהֵבֵאתִי עָלַי קְלָלָה וְלֹא בְרָכָה:
(יג) וַתֹּאמֶר לוֹ אִמּוֹ עָלַי קִלְלָתְךָ בְּנִי אַךְ שְׁמַע בְּקֹלִי וְלֵךְ קַח לִי:
(יד) וַיֵּלֶךְ וַיִּקַּח וַיָּבֵא לְאִמּוֹ וַתַּעַשׂ אִמּוֹ מַטְעַמִּים כַּאֲשֶׁר אָהֵב אָבִיו:
(טו) וַתִּקַּח רִבְקָה אֶת בִּגְדֵי עֵשָׂו בְּנָהּ הַגָּדֹל הַחֲמֻדֹת אֲשֶׁר אִתָּהּ בַּבָּיִת וַתַּלְבֵּשׁ אֶת יַעֲקֹב בְּנָהּ הַקָּטָן:
(טז) וְאֵת עֹרֹת גְּדָיֵי הָעִזִּים הִלְבִּישָׁה עַל יָדָיו וְעַל חֶלְקַת צַוָּארָיו:
(יז) וַתִּתֵּן אֶת הַמַּטְעַמִּים וְאֶת הַלֶּחֶם אֲשֶׁר עָשָׂתָה בְּיַד יַעֲקֹב בְּנָהּ:
(יח) וַיָּבֹא אֶל אָבִיו וַיֹּאמֶר אָבִי וַיֹּאמֶר הִנֶּנִּי מִי אַתָּה בְּנִי:
(יט) וַיֹּאמֶר יַעֲקֹב אֶל אָבִיו אָנֹכִי עֵשָׂו בְּכֹרֶךָ עָשִׂיתִי כַּאֲשֶׁר דִּבַּרְתָּ אֵלָי קוּם נָא שְׁבָה וְאָכְלָה מִצֵּידִי בַּעֲבוּר תְּבָרְכַנִּי נַפְשֶׁךָ:
(כ) וַיֹּאמֶר יִצְחָק אֶל בְּנוֹ מַה זֶּה מִהַרְתָּ לִמְצֹא בְּנִי וַיֹּאמֶר כִּי הִקְרָה יְהוָה אֱלֹהֶיךָ לְפָנָי:
(כא) וַיֹּאמֶר יִצְחָק אֶל יַעֲקֹב גְּשָׁה נָּא וַאֲמֻשְׁךָ בְּנִי הַאַתָּה זֶה בְּנִי עֵשָׂו אִם לֹא:
(כב) וַיִּגַּשׁ יַעֲקֹב אֶל יִצְחָק אָבִיו וַיְמֻשֵּׁהוּ וַיֹּאמֶר הַקֹּל קוֹל יַעֲקֹב וְהַיָּדַיִם יְדֵי עֵשָׂו:
(כג) וְלֹא הִכִּירוֹ כִּי הָיוּ יָדָיו כִּידֵי עֵשָׂו אָחִיו שְׂעִרֹת וַיְבָרְכֵהוּ:
(כד) וַיֹּאמֶר אַתָּה זֶה בְּנִי עֵשָׂו וַיֹּאמֶר אָנִי:
(כה) וַיֹּאמֶר הַגִּשָׁה לִּי וְאֹכְלָה מִצֵּיד בְּנִי לְמַעַן תְּבָרֶכְךָ נַפְשִׁי וַיַּגֶּשׁ לוֹ וַיֹּאכַל וַיָּבֵא לוֹ יַיִן וַיֵּשְׁתְּ:
(כו) וַיֹּאמֶר אֵלָיו יִצְחָק אָבִיו גְּשָׁה נָּא וּשְׁקָה לִּי בְּנִי:
(כז) וַיִּגַּשׁ וַיִּשַּׁק לוֹ וַיָּרַח אֶת רֵיחַ בְּגָדָיו וַיְבָרְכֵהוּ וַיֹּאמֶר רְאֵה רֵיחַ בְּנִי כְּרֵיחַ שָׂדֶה אֲשֶׁר בֵּרְכוֹ יְהוָה:
(כח) וְיִתֶּן לְךָ הָאֱלֹהִים מִטַּל הַשָּׁמַיִם וּמִשְׁמַנֵּי הָאָרֶץ וְרֹב דָּגָן וְתִירֹשׁ:
(כט) יַעַבְדוּךָ עַמִּים וְיִשְׁתַּחֲווּ לְךָ לְאֻמִּים הֱוֵה גְבִיר לְאַחֶיךָ וְיִשְׁתַּחֲווּ לְךָ בְּנֵי אִמֶּךָ אֹרְרֶיךָ אָרוּר וּמְבָרְכֶיךָ בָּרוּךְ:
(ל) וַיְהִי כַּאֲשֶׁר כִּלָּה יִצְחָק לְבָרֵךְ אֶת יַעֲקֹב וַיְהִי אַךְ יָצֹא יָצָא יַעֲקֹב מֵאֵת פְּנֵי יִצְחָק אָבִיו וְעֵשָׂו אָחִיו בָּא מִצֵּידוֹ:
(לא) וַיַּעַשׂ גַּם הוּא מַטְעַמִּים וַיָּבֵא לְאָבִיו וַיֹּאמֶר לְאָבִיו יָקֻם אָבִי וְיֹאכַל מִצֵּיד בְּנוֹ בַּעֲבֻר תְּבָרְכַנִּי נַפְשֶׁךָ:
(לב) וַיֹּאמֶר לוֹ יִצְחָק אָבִיו מִי אָתָּה וַיֹּאמֶר אֲנִי בִנְךָ בְכֹרְךָ עֵשָׂו:
(לג) וַיֶּחֱרַד יִצְחָק חֲרָדָה גְדֹלָה עַד מְאֹד וַיֹּאמֶר מִי אֵפוֹא הוּא הַצָּד צַיִד וַיָּבֵא לִי וָאֹכַל מִכֹּל בְּטֶרֶם תָּבוֹא וָאֲבָרְכֵהוּ גַּם בָּרוּךְ יִהְיֶה:
(לד) כִּשְׁמֹעַ עֵשָׂו אֶת דִּבְרֵי אָבִיו וַיִּצְעַק צְעָקָה גְדֹלָה וּמָרָה עַד מְאֹד וַיֹּאמֶר לְאָבִיו בָּרְכֵנִי גַם אָנִי אָבִי:
(לה) וַיֹּאמֶר בָּא אָחִיךָ בְּמִרְמָה וַיִּקַּח בִּרְכָתֶךָ:
(לו) וַיֹּאמֶר הֲכִי קָרָא שְׁמוֹ יַעֲקֹב וַיַּעְקְבֵנִי זֶה פַעֲמַיִם אֶת בְּכֹרָתִי לָקָח וְהִנֵּה עַתָּה לָקַח בִּרְכָתִי וַיֹּאמַר הֲלֹא אָצַלְתָּ לִּי בְּרָכָה:
(לז) וַיַּעַן יִצְחָק וַיֹּאמֶר לְעֵשָׂו הֵן גְּבִיר שַׂמְתִּיו לָךְ וְאֶת כָּל אֶחָיו נָתַתִּי לוֹ לַעֲבָדִים וְדָגָן וְתִירֹשׁ סְמַכְתִּיו וּלְכָה אֵפוֹא מָה אֶעֱשֶׂה בְּנִי:
(לח) וַיֹּאמֶר עֵשָׂו אֶל אָבִיו הַבְרָכָה אַחַת הִוא לְךָ אָבִי בָּרְכֵנִי גַם אָנִי אָבִי וַיִּשָּׂא עֵשָׂו קֹלוֹ וַיֵּבְךְּ:
(לט) וַיַּעַן יִצְחָק אָבִיו וַיֹּאמֶר אֵלָיו הִנֵּה מִשְׁמַנֵּי הָאָרֶץ יִהְיֶה מוֹשָׁבֶךָ וּמִטַּל הַשָּׁמַיִם מֵעָל:
(מ) וְעַל חַרְבְּךָ תִחְיֶה וְאֶת אָחִיךָ תַּעֲבֹד וְהָיָה כַּאֲשֶׁר תָּרִיד וּפָרַקְתָּ עֻלּוֹ מֵעַל צַוָּארֶךָ:
(מא) וַיִּשְׂטֹם עֵשָׂו אֶת יַעֲקֹב עַל הַבְּרָכָה אֲשֶׁר בֵּרְכוֹ אָבִיו וַיֹּאמֶר עֵשָׂו בְּלִבּוֹ יִקְרְבוּ יְמֵי אֵבֶל אָבִי וְאַהַרְגָה אֶת יַעֲקֹב אָחִי:
(מב) וַיֻּגַּד לְרִבְקָה אֶת דִּבְרֵי עֵשָׂו בְּנָהּ הַגָּדֹל וַתִּשְׁלַח וַתִּקְרָא לְיַעֲקֹב בְּנָהּ הַקָּטָן וַתֹּאמֶר אֵלָיו הִנֵּה עֵשָׂו אָחִיךָ מִתְנַחֵם לְךָ לְהָרְגֶךָ:
(מג) וְעַתָּה בְנִי שְׁמַע בְּקֹלִי וְקוּם בְּרַח לְךָ אֶל לָבָן אָחִי חָרָנָה:
(מד) וְיָשַׁבְתָּ עִמּוֹ יָמִים אֲחָדִים עַד אֲשֶׁר תָּשׁוּב חֲמַת אָחִיךָ:
(מה) עַד שׁוּב אַף אָחִיךָ מִמְּךָ וְשָׁכַח אֵת אֲשֶׁר עָשִׂיתָ לּוֹ וְשָׁלַחְתִּי וּלְקַחְתִּיךָ מִשָּׁם לָמָה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָד:
(מו) וַתֹּאמֶר רִבְקָה אֶל יִצְחָק קַצְתִּי בְחַיַּי מִפְּנֵי בְּנוֹת חֵת אִם לֹקֵחַ יַעֲקֹב אִשָּׁה מִבְּנוֹת חֵת כָּאֵלֶּה מִבְּנוֹת הָאָרֶץ לָמָּה לִּי חַיִּים:

## תקציר

יצחק זקן ועיניו כהות מלראות. הוא קורא לעשו בנו הגדול ומבקש ממנו לצאת לשדה, לצוד ציד ולהכין לו מטעמים כדי שיברכו לפני מותו. רבקה שומעת את הדברים ופונה ליעקב בנה. היא מורה לו להביא שני גדיי עיזים מהצאן, והיא תכין מטעמים ליצחק כדי שיעקב יקבל את הברכה במקום עשו. יעקב חושש שאביו ימשש אותו ויגלה את ההונאה, כי עשו איש שעיר ויעקב איש חלק, ואז יביא על עצמו קללה ולא ברכה. רבקה אומרת "עלי קללתך בני" ומצווה עליו לשמוע בקולה. יעקב מביא את הגדיים, רבקה מכינה מטעמים, מלבישה את יעקב בבגדי עשו החמודות, ומכסה את ידיו ואת חלקת צווארו בעורות גדיי העיזים. יעקב נכנס אל אביו ואומר "אנכי עשו בכורך". יצחק חושד ושואל כיצד מצא כל כך מהר, ויעקב עונה "כי הקרה ה' אלהיך לפני". יצחק מבקש למשש אותו ואומר את המשפט המפורסם "הקול קול יעקב והידיים ידי עשו" - אך לא מכיר אותו כי ידיו שעירות, ומברכו. הברכה כוללת טל השמים, משמני הארץ, רוב דגן ותירוש, שעמים יעבדוהו ולאומים ישתחוו לו, שיהיה גביר לאחיו, ושאוררו יהיה ארור ומברכו ברוך. מיד לאחר שיעקב יוצא, עשו חוזר מהציד, מכין מטעמים ונכנס לאביו. יצחק חורד חרדה גדולה כשמבין מה קרה ואומר "גם ברוך יהיה" - מאשר את ברכת יעקב. עשו צועק צעקה גדולה ומרה ומבקש ברכה גם הוא. עשו אומר "הכי קרא שמו יעקב ויעקבני זה פעמיים" - את הבכורה לקח ועתה גם את הברכה. יצחק נותן לעשו ברכה פחותה: מושבו יהיה משמני הארץ ומטל השמים, על חרבו יחיה, את אחיו יעבוד, אך כאשר ירד - יפרוק עולו מעל צווארו. עשו שונא את יעקב ומתכנן להרגו לאחר ימי אבל אביו. רבקה שומעת ומצווה על יעקב לברוח אל לבן אחיה לחרן, ומצדיקה את שליחתו בפני יצחק בטענה שהיא קצה בחייה מפני בנות חת ואינה רוצה שיעקב יישא אישה מבנות הארץ.

## דמויות מרכזיות

- **יצחק** - אב זקן ועיוור, רוצה לברך את עשו בנו האהוב לפני מותו. חושד ביעקב אך מתרצה. כשמגלה את התרמית חורד חרדה גדולה אך מאשר את הברכה ("גם ברוך יהיה"). נותן לעשו ברכה פחותה.
- **רבקה** - היא היוזמת והמתכננת של כל התרמית. שומעת את דברי יצחק לעשו, מורה ליעקב כיצד לפעול, מכינה את המטעמים, מלבישה אותו בבגדי עשו ובעורות הגדיים. לוקחת על עצמה את הקללה ("עלי קללתך בני"). בסוף מציעה לשלוח את יעקב ללבן כדי להצילו מנקמת עשו.
- **יעקב** - מבצע את התרמית לפי הוראות אמו. חושש בהתחלה מגילוי ומקללה. משקר לאביו ואומר "אנכי עשו בכורך" ואף מזכיר את שם ה' ("כי הקרה ה' אלהיך לפני"). מקבל את ברכת הבכורה ונאלץ לברוח לחרן.
- **עשו** - יוצא לצוד לפי בקשת אביו, חוזר ומגלה שברכתו נגנבה. צועק צעקה גדולה ומרה, מבקש ברכה מאביו, מקבל ברכה פחותה. מתכנן להרוג את יעקב לאחר מות אביו. מתאר את יעקב כמי ש"עקבו" פעמיים - בבכורה ובברכה.

## מקומות ומה קרה בהם

- **אוהל יצחק / בית יצחק** - כל אירועי הפרק מתרחשים בתוך המגורים של יצחק. שם הוא יושב זקן ועיוור, שם מברך את יעקב (בטעות) ואת עשו, ושם מתגלה התרמית.
- **השדה** - לשם יוצא עשו לצוד ציד לאביו. השדה מופיע גם בברכה: "ראה ריח בני כריח שדה אשר ברכו ה'".
- **הצאן** - משם מביא יעקב שני גדיי עיזים בהוראת רבקה.
- **חרן** - היעד שאליו רבקה שולחת את יעקב לברוח, אל לבן אחיה, מפני עשו.

## מספרים ומה הם מייצגים

- **2 גדיי עיזים** - שני הגדיים שרבקה מבקשת מיעקב להביא מהצאן להכנת המטעמים (פסוק ט).
- **2 ברכות** - ברכה ראשונה (עיקרית) ליעקב וברכה שנייה (פחותה) לעשו.
- **2 פעמים** - עשו טוען שיעקב עקב אותו פעמיים: פעם אחת בלקיחת הבכורה ופעם שנייה בלקיחת הברכה (פסוק לו).
- **2 בנים** - רבקה אומרת "למה אשכל גם שניכם יום אחד" - חוששת לאבד את שני בניה (אם עשו יהרוג את יעקב, עשו ייענש במוות).
- **יום אחד** - רבקה חוששת לאבד את שני בניה ביום אחד (פסוק מה).

## מילות מפתח

- **ותכהין עיניו מראות** - עיוורונו של יצחק, התנאי שמאפשר את כל התרמית.
- **מטעמים** - המאכל המיוחד שיצחק מבקש כתנאי לברכה. מילה חוזרת בפרק (פסוקים ד, ז, ט, יד, יז, לא).
- **צידה / ציד** - מה שיצחק מבקש מעשו לצוד. מילה מנחה בפרק.
- **ברכה / ברך** - המילה המרכזית בפרק, חוזרת פעמים רבות. הברכה היא מוקד הסיפור כולו.
- **מרמה** - יצחק אומר לעשו "בא אחיך במרמה ויקח ברכתך" (פסוק לה).
- **איש שעיר / איש חלק** - הניגוד בין עשו ליעקב, שמחייב את ההתחפשות.
- **בגדי עשו החמודות** - הבגדים שרבקה מלבישה את יעקב כדי לרמות את יצחק (ריח השדה).
- **עורות גדיי העיזים** - כיסוי ידיו וצווארו של יעקב כדי לחקות את שעירות עשו.
- **קול / ידיים** - "הקול קול יעקב והידיים ידי עשו" - הניגוד בין שתי החושים של יצחק.
- **צעקה גדולה ומרה** - תגובת עשו כששומע שברכתו נלקחה (פסוק לד).
- **ויעקבני** - משחק מילים של עשו על שם יעקב, שמשמעותו "עקב" או "רימה" (פסוק לו).
- **גביר לאחיך** - חלק מברכת יעקב, שיהיה אדון לאחיו (פסוק כט).
- **על חרבך תחיה** - חלק מברכת עשו, חיי מלחמה (פסוק מ).
- **ופרקת עולו** - ההבטחה לעשו שבבוא הזמן יפרוק את עול אחיו מעליו (פסוק מ).
- **ימי אבל אבי** - עשו מתכנן להרוג את יעקב רק לאחר מות יצחק (פסוק מא).
- **ימים אחדים** - רבקה אומרת ליעקב לשהות בחרן "ימים אחדים" - אך בפועל ישהה עשרים שנה.
- **קצתי בחיי מפני בנות חת** - התירוץ של רבקה ליצחק לשליחת יעקב לחרן (פסוק מו).

## פסוקים לזיהוי

1. **פסוק כ"ז:א** - "וַיְהִי כִּי זָקֵן יִצְחָק וַתִּכְהֶיןָ עֵינָיו מֵרְאֹת וַיִּקְרָא אֶת עֵשָׂו בְּנוֹ הַגָּדֹל" - פסוק הפתיחה, מציג את המצב: יצחק זקן ועיוור.

2. **פסוק כ"ז:יא-יב** - "הֵן עֵשָׂו אָחִי אִישׁ שָׂעִר וְאָנֹכִי אִישׁ חָלָק. אוּלַי יְמֻשֵּׁנִי אָבִי וְהָיִיתִי בְעֵינָיו כִּמְתַעְתֵּעַ וְהֵבֵאתִי עָלַי קְלָלָה וְלֹא בְרָכָה" - חששו של יעקב מגילוי ההונאה.

3. **פסוק כ"ז:יג** - "וַתֹּאמֶר לוֹ אִמּוֹ עָלַי קִלְלָתְךָ בְּנִי אַךְ שְׁמַע בְּקֹלִי וְלֵךְ קַח לִי" - רבקה לוקחת על עצמה את הקללה.

4. **פסוק כ"ז:יט** - "וַיֹּאמֶר יַעֲקֹב אֶל אָבִיו אָנֹכִי עֵשָׂו בְּכֹרֶךָ עָשִׂיתִי כַּאֲשֶׁר דִּבַּרְתָּ אֵלָי קוּם נָא שְׁבָה וְאָכְלָה מִצֵּידִי בַּעֲבוּר תְּבָרְכַנִּי נַפְשֶׁךָ" - יעקב מתחזה לעשו ומשקר לאביו.

5. **פסוק כ"ז:כ** - "וַיֹּאמֶר יִצְחָק אֶל בְּנוֹ מַה זֶּה מִהַרְתָּ לִמְצֹא בְּנִי וַיֹּאמֶר כִּי הִקְרָה יְהוָה אֱלֹהֶיךָ לְפָנָי" - יצחק חושד ויעקב מזכיר את שם ה' בשקר.

6. **פסוק כ"ז:כב** - "וַיִּגַּשׁ יַעֲקֹב אֶל יִצְחָק אָבִיו וַיְמֻשֵּׁהוּ וַיֹּאמֶר הַקֹּל קוֹל יַעֲקֹב וְהַיָּדַיִם יְדֵי עֵשָׂו" - הפסוק המפורסם ביותר בפרק, יצחק מזהה את הקול כשל יעקב אך הידיים כשל עשו.

7. **פסוק כ"ז:כז-כט** - ברכת יעקב: "רְאֵה רֵיחַ בְּנִי כְּרֵיחַ שָׂדֶה אֲשֶׁר בֵּרְכוֹ יְהוָה. וְיִתֶּן לְךָ הָאֱלֹהִים מִטַּל הַשָּׁמַיִם וּמִשְׁמַנֵּי הָאָרֶץ וְרֹב דָּגָן וְתִירֹשׁ. יַעַבְדוּךָ עַמִּים וְיִשְׁתַּחֲווּ לְךָ לְאֻמִּים הֱוֵה גְבִיר לְאַחֶיךָ וְיִשְׁתַּחֲווּ לְךָ בְּנֵי אִמֶּךָ אֹרְרֶיךָ אָרוּר וּמְבָרְכֶיךָ בָּרוּךְ" - הברכה המרכזית: טל השמים, משמני הארץ, דגן ותירוש, שלטון על עמים ואחים.

8. **פסוק כ"ז:לג** - "וַיֶּחֱרַד יִצְחָק חֲרָדָה גְדֹלָה עַד מְאֹד וַיֹּאמֶר מִי אֵפוֹא הוּא הַצָּד צַיִד וַיָּבֵא לִי וָאֹכַל מִכֹּל בְּטֶרֶם תָּבוֹא וָאֲבָרְכֵהוּ גַּם בָּרוּךְ יִהְיֶה" - חרדת יצחק כשמגלה את התרמית, ואישורו שהברכה תעמוד.

9. **פסוק כ"ז:לד** - "כִּשְׁמֹעַ עֵשָׂו אֶת דִּבְרֵי אָבִיו וַיִּצְעַק צְעָקָה גְדֹלָה וּמָרָה עַד מְאֹד וַיֹּאמֶר לְאָבִיו בָּרְכֵנִי גַם אָנִי אָבִי" - הצעקה הגדולה והמרה של עשו, רגע דרמטי מרכזי.

10. **פסוק כ"ז:לה** - "וַיֹּאמֶר בָּא אָחִיךָ בְּמִרְמָה וַיִּקַּח בִּרְכָתֶךָ" - יצחק מסביר לעשו מה קרה.

11. **פסוק כ"ז:לו** - "וַיֹּאמֶר הֲכִי קָרָא שְׁמוֹ יַעֲקֹב וַיַּעְקְבֵנִי זֶה פַעֲמַיִם אֶת בְּכֹרָתִי לָקָח וְהִנֵּה עַתָּה לָקַח בִּרְכָתִי" - עשו מנתח את שמו של יעקב כ"עוקב", וטוען שנרמה פעמיים.

12. **פסוק כ"ז:לח** - "הַבְרָכָה אַחַת הִוא לְךָ אָבִי בָּרְכֵנִי גַם אָנִי אָבִי וַיִּשָּׂא עֵשָׂו קֹלוֹ וַיֵּבְךְּ" - בכיו של עשו, שואל אם יש רק ברכה אחת.

13. **פסוק כ"ז:לט-מ** - ברכת עשו: "הִנֵּה מִשְׁמַנֵּי הָאָרֶץ יִהְיֶה מוֹשָׁבֶךָ וּמִטַּל הַשָּׁמַיִם מֵעָל. וְעַל חַרְבְּךָ תִחְיֶה וְאֶת אָחִיךָ תַּעֲבֹד וְהָיָה כַּאֲשֶׁר תָּרִיד וּפָרַקְתָּ עֻלּוֹ מֵעַל צַוָּארֶךָ" - ברכה פחותה: חיי חרב, עבדות לאח, אך הבטחה לשחרור עתידי.

14. **פסוק כ"ז:מא** - "וַיִּשְׂטֹם עֵשָׂו אֶת יַעֲקֹב עַל הַבְּרָכָה אֲשֶׁר בֵּרְכוֹ אָבִיו וַיֹּאמֶר עֵשָׂו בְּלִבּוֹ יִקְרְבוּ יְמֵי אֵבֶל אָבִי וְאַהַרְגָה אֶת יַעֲקֹב אָחִי" - שנאת עשו ותכנית הנקמה.

15. **פסוק כ"ז:מה** - "עַד שׁוּב אַף אָחִיךָ מִמְּךָ וְשָׁכַח אֵת אֲשֶׁר עָשִׂיתָ לּוֹ וְשָׁלַחְתִּי וּלְקַחְתִּיךָ מִשָּׁם לָמָה אֶשְׁכַּל גַּם שְׁנֵיכֶם יוֹם אֶחָד" - רבקה מבטיחה ליעקב שתשלח לקחתו כשיירגע עשו, וחוששת לאבד את שני בניה.

16. **פסוק כ"ז:מו** - "וַתֹּאמֶר רִבְקָה אֶל יִצְחָק קַצְתִּי בְחַיַּי מִפְּנֵי בְּנוֹת חֵת אִם לֹקֵחַ יַעֲקֹב אִשָּׁה מִבְּנוֹת חֵת כָּאֵלֶּה מִבְּנוֹת הָאָרֶץ לָמָּה לִּי חַיִּים" - התירוץ הרשמי של רבקה ליצחק לשליחת יעקב.
